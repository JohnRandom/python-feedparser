Getting started

Version 4.0 is under active development and already fully supports Atom 1.0 feeds. It will also eventually support Apple's iTunes extensions, Yahoo's Media RSS extensions, and Amazon's OpenSearch extensions. If you are comfortable running undocumented pre-release software, you can test feedparser.py CVS. Please report bugs via SourceForge.

>>> import feedparser
>>> d = feedparser.parse("http://feedparser.org/tests/examples/atom.xml")
>>> d['feed']['title']             # feed data is a dictionary
u'Sample Feed'
>>> d.feed.title                   # get values attr-style or dict-style
u'Sample Feed'
>>> d.channel.title                # use RSS or Atom terminology anywhere
u'Sample Feed'
>>> d.feed.link                    # resolves relative links
u'http://example.org/'
>>> d.feed.tagline                 # parses escaped HTML
u'For documentation <em>only</em>'
>>> d.channel.description          # RSS terminology works here too
u'For documentation <em>only</em>'
>>> d.feed.info                    # parses inline XHTML
u'<div><p>This is an Atom syndication feed.</p></div>'
>>> len(d['entries'])              # entries are a list
1
>>> d['entries'][0]['title']       # each entry is a dictionary
u'First entry title'
>>> d.entries[0].title             # attr-style works here too
u'First entry title'
>>> d['items'][0].title            # RSS terminology works here too
u'First entry title'
>>> e = d.entries[0]
>>> e.link                         # easy access to alternate link
u'http://example.org/entry/3'
>>> e.links[1].rel                 # full access to all Atom links
u'service.edit'
>>> e.links[1].href                # resolves relative links here too
u'http://example.org/api/edit/3'
>>> e.author_detail.name           # author data is a dictionary
u'Mark Pilgrim'
>>> e.modified_parsed              # parses all date formats
(2004, 4, 20, 11, 56, 34, 1, 111, 0)
>>> e.content[0].value             # sanitizes dangerous HTML
u'<div>Watch out for <em>nasty tricks</em></div>'
>>> d.version                      # reports feed type and version
u'atom03'
>>> d.encoding                     # auto-detects character encoding
u'iso-8859-1'
>>> d.headers.get('Content-type')  # full access to all HTTP headers
u'application/xml'
